Abstraccion de objetos:
- Usuario
	id int
	name varchar
	email varchar
	username varchar
	password
	celular int
	activo bool
	rol int

- Computadora
	id int	
	disponibilidad int
	personalizada boolean

- Ensamblaje
	id int 
	idComputadora int
	idEmpleado int
	estado varchar (en proceso, ensamblado, vendido)

- Componente
	id int
	tipoComponente varchar	(procesador, memoriaRAM, almacenamiento, fuenteDePoder, gabinete)
	marca varchar			(todos)
	consumoEnergetico int 	(procesador, ram, almacenamiento)
	nucleos int 			(procesador)
	velocidad decimal(10,2) (procesador, ram)
	capacidad int 			(ram, almacenamiento)
	tipo varchar 			(ram, almacenamiento)
	potencia int 			(fuenteDePoder)
	eficiencia varchar		(fuenteDePoder)
	color varchar			(gabinete)
	stock int				(todos)
	precio decimal(10,2)	(todos)

- ComponenteComputadora
	- idComputadora int
	- idComponente int
	- cantidad int

- Venta
	id int
	total int
	nitUsuario int
	nitEmpleado int

- DetalleVenta
	id int
	idVenta int
	idComputadora int
	idComponente int
	cantidad int
	subtotal int

- Factura
	id int
	idVenta int	
	nit int (nit del cliente)
	fecha date

- DetalleEnsamblaje
	idEnsamblaje int
	idComponente int
	idComponente int

- ReporteCliente
	id int
	nitUsuario int
	tipo (queja o sugerencia) varchar
	idEmpleado int (si hay un empleado involucrado)
	idComponente int (si hay un componente involucrado)
	idComputadora int (si hay una computadora involucrada)
	descripcion varchar (200)
